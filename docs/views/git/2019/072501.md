```bash
# 查看所有标签
git tag
git tag -l

# 添加标签
git tag -a v1.0.0 -m '测试'

# 提交 tag 到远程仓库
git push origin v1.0.0
# 等同于如下操作：  
git push origin refs/tags/<源标签名>:refs/tags/<目的标签名>

# 提交所有 tag 到远程仓库
git push origin --tags

# 删除标签
git tag -d v1.0.0

# 删除远程标签
# 
git push origin :refs/tags/v1.0.0
```
