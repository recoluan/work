## 方法一

```js
function isEmptyObject (obj) {
  for (let key in obj){
    return false
  }　　
  return true
}
```

## 方法一进阶

继承的属性不算。

```js
function isOwnEmpty (obj) {
  for (let name in obj) {
    if (obj.hasOwnProperty(name)) {
      return false
    }
  }
  return true
}
```

```js
function isEmptyObject (obj) {
  if (Object.getOwnPropertyNames(obj).length ===0) {
    return true
  }
  return false
}
```
